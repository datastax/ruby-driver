Result paging

Cassandra native protocol v2, used by Apache Cassandra 2.0, allows paging through query results.

Page size can be specified by passing :page_size option to Cassandra::Session#execute.

Once a Cassandra::Result has been received, it can be paged through using Cassandra::Result#next_page or Cassandra::Result#next_page_async methods for synchronous and asynchronous next page retrieval accordingly.

Given
a running cassandra cluster with schema:
CREATE KEYSPACE simplex WITH replication = {'class': 'SimpleStrategy', 'replication_factor': 3};
CREATE TABLE simplex.test (k text, v int, PRIMARY KEY (k, v));

INSERT INTO simplex.test (k, v) VALUES ('b', 1);

Paging through results synchronously

Given
the following example:
require 'cassandra'

cluster = Cassandra.cluster
session = cluster.connect("simplex")
result  = session.execute("SELECT * FROM test", page_size: 5)

loop do
  puts "last page? #{result.last_page?}"
  puts "page size: #{result.size}"

  result.each do |row|
    puts row
  end
  puts ""

  break if result.last_page?
  result = result.next_page
end

Paging through results asynchronously

Given
the following example:
require 'cassandra'

def page_through(future)
  future.then do |result|
    puts "last page? #{result.last_page?}"
    puts "page size: #{result.size}"

    result.each do |row|
      puts row
    end
    puts ""

    page_through(result.next_page_async) unless result.last_page?
  end
end

cluster = Cassandra.cluster
session = cluster.connect("simplex")
select  = session.prepare("SELECT * FROM test")
future  = session.execute_async(select, page_size: 5)

page_through(future).join

Using paging state for stateless paging

Given
the following example:
require 'cassandra'

cluster      = Cassandra.cluster
session      = cluster.connect("simplex")
paging_state = nil

loop do
  result = session.execute("SELECT * FROM test", page_size: 5, paging_state: paging_state)
  puts "last page? #{result.last_page?}"
  puts "page size: #{result.size}"

  result.each do |row|
    puts row
  end
  puts ""

  break if result.last_page?
  paging_state = result.paging_state
end
